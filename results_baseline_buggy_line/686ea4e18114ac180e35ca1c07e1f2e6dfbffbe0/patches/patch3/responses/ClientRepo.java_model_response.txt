    public void purgeBot(UUID botId) throws IOException {
        State state = sf.create(botId);
        if (state == null)
            return;

        boolean purged = state.removeState();
        if (!purged)
            throw new IOException("Failed to purge Bot: " + botId);
    }

    public Client getHttpClient() {

    }
