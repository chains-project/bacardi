```java
package com.maxmind.minfraud.request;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.maxmind.minfraud.AbstractModel;
import org.apache.commons.codec.binary.Hex; // Updated import for newer versions
import java.nio.charset.StandardCharsets; // Added for encoding to bytes
import java.security.MessageDigest; // Added for digest computation
import java.security.NoSuchAlgorithmException; // Added for handling exceptions

/**
 * Account related data for the minFraud request
 */
public final class Account extends AbstractModel {
    private final String userId;
    private final String usernameMd5;

    private Account(Account.Builder builder) {
        userId = builder.userId;
        usernameMd5 = builder.usernameMd5;
    }

    /**
     * {@code Builder} creates instances of {@code Account}
     * from values set by the builder's methods.
     */
    public static final class Builder {
        String userId;
        String usernameMd5;

        /**
         * @param id A unique user ID associated with the end-user in your
         *           system. If your system allows the login name for the
         *           account to be changed, this should not be the login
         *           name for the account, but rather should be an internal
         *           ID that does not change. This is not your MaxMind user
         *           ID.
         * @return The builder object.
         */
        public Account.Builder userId(String id) {
            this.userId = id;
            return this;
        }

        /**
         * @param username The username associated with the account. This is
         *                 <em>not</em> the MD5 of username. This method
         *                 automatically runs {@code DigestUtils.md5Hex}
         *                 on the string passed to it.
         * @return The builder object.
         */
        public Account.Builder username(String username) {
            this.usernameMd5 = computeMD5(username); // Use custom method to compute MD5
            return this;
        }

        private String computeMD5(String input) {
            try {
                MessageDigest md = MessageDigest.getInstance("MD5");
                byte[] digest = md.digest(input.getBytes(StandardCharsets.UTF_8));
                return Hex.encodeHexString(digest); // Convert byte array to hex string
            } catch (NoSuchAlgorithmException e) {
                throw new RuntimeException(e); // Handle the exception appropriately
            }
        }

        /**
         * @return An instance of {@code Account} created from the
         * fields set on this builder.
         */
        public Account build() {
            return new Account(this);
        }
    }

    /**
     * @return The user ID.
     */
    @JsonProperty("user_id")
    public String getUserId() {
        return userId;
    }

    /**
     * @return The MD5 of the username passed to the builder.
     */
    @JsonProperty("username_md5")
    public String getUsernameMd5() {
        return usernameMd5;
    }
}
``` 

### Changes Made:
1. Updated the import statement to use `org.apache.commons.codec.binary.Hex` for hex encoding, suitable for newer versions.
2. Replaced the use of `DigestUtils.md5Hex` with a custom method `computeMD5` that manually computes the MD5 hash using `MessageDigest` and converts it to a hex string.
3. Added error handling for the `NoSuchAlgorithmException` that may occur when getting the MD5 instance.

This ensures that the code is compatible with the updated dependencies and retains the original functionality.
