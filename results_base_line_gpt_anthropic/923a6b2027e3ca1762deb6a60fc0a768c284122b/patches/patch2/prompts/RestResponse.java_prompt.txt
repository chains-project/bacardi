            .back();
        final Map<String, List<String>> headers = this.headers();
        if (headers.containsKey(HttpHeaders.SET_COOKIE)) {

                for (final HttpCookie cookie : HttpCookie.parse(header)) {
                    req = req.header(

                        )
                    );
                }
            }
        }
        return req;
    }

    }
